\section*{Graphs}
$G=(V,E)$, $|V|=n$, $|E|=m$.

\textbf{Path}: sequence of vertices with edges between. \textbf{Cycle}: path where start=end.

\textbf{Connected}: can reach any vertex from any other. \textbf{Strongly conn} (dir): can go $u\to v$ and $v\to u$ for all pairs. \textbf{Weakly}: connected if you ignore arrow directions.

\textbf{Bipartite}: split $V$ into 2 sets, edges only between sets. No odd cycles.

\textbf{Acyclic}: no cycles. \textbf{DAG}: directed + acyclic.

\bbox{$\sum \deg(v)=2m$} \quad Max edges: $\binom{n}{2}$, dir: $n(n-1)$

\subsection*{Representations}
\textbf{Adj Matrix}: $A_{ij}=1$ iff edge. $O(n^2)$ space, $O(1)$ edge, $O(n)$ neighbors.

\textbf{Adj List}: store neighbor lists. $O(n+m)$ space, $O(\deg)$ edge, $O(\deg)$ neighbors.

\subsection*{BST}
Left $<$ node $\leq$ right. Search/insert/delete: $O(h)$ where $h$=tree height (longest root-to-leaf path). Balanced: $O(\log n)$, worst: $O(n)$.
